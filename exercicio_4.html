Exercício 4

Faça um programa que receba um número e usando laços de repetição calcule e mostre a tabuada desse número.

Código feito em linguagem C
Tabuada

#include <stdio.h>
#include <stdlib.h>
#include <locale.h>//Biblioteca para identificar caracteres especiais.

void limpa_tela(void);//identifica o sistema operacional e limpa a tela conforme o sistema.

int main()
{
int numero, resposta;
int contador=0;

setlocale(LC_ALL,"");//Para identificar caracteres especiais.

do//Loop
{
limpa_tela();//identifica o sistema operacional e limpa a tela conforme o sistema.
printf("Tabuada\n\n");
printf("Digite o número da tabuada: ");
scanf("%d", &numero);
getchar();
for(contador=0; contador<=10; contador++)//ele vai fazer o laço de repetição até 10x
{
printf("%d x %d = %d\n", contador, numero, contador*numero);
}
printf("\n1- Novo calculo\n");
printf("2- Sair\n\n");
scanf("%d", &resposta);
getchar();
}
while(resposta != 2);//Condição para sair do loop
{
system("pause");
return 0;
}
}

void limpa_tela(void)//identifica o sistema operacional e limpa a tela conforme o sistema
{
// Limpa tela para sys windows
#ifdef _WIN32
system("cls");
// Limpa tela para sys linux e apple
#elif defined __linux__ || defined __APPLE__
system("clear");
// Se não for nenhum desses dois vai aparecer essa mensagem
#else
#pragma message ("So não detectado!")
#endif
}
